Taking input from Command Line argument in 1D Array

class OneDArray {
  public static void main(String x[])
  {
   int arr[]; //Array declaration
   arr=new int[5];
   for(int i=0;i<5;i++)
   {
    arr[i]=Integer.parseInt(x[i]);
   }
   System.out.println("Display the array elements");
   for(int i=0;i<5;i++)
   {
    System.out.println(arr[i]);
   }
  }
 }
